- PN: {ISBN}
Forfatter(FID, fornavn, etternavn, fødselsdato)

Forlag(navn, FID)
- KN: {navn}
- PN: {navn}
Kapittel(nummer, ISBN, tittel, første, siste)

- PN: {nummer, ISBN}

- PN: {ISBN, FID}

# Fremmednøkler

Kapittel(ISBN) -> Bok(ISBN)
Forlag(FID) -> Forfatter(FID)
Bok(forlag) -> Forlag(navn)
Skrevet(ISBN) -> Bok(ISBN)
Skrevet(FID) -> Forfatter(FID)
Sjanger(ISBN) -> Bok(ISBN)
